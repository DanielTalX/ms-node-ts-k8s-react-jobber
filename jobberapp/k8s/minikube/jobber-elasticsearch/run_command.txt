- create production namespace if it does not exist
kubectl create namespace production

- add all at the same time:
kubectl apply -f .

- add them individually
kubectl apply -f storageClass.yaml
kubectl apply -f pv.yaml
kubectl apply -f service.yaml
kubectl apply -f elasticsearch.yaml

- display logs of a pod
kubectl logs jobber-redis-0 -n production

- verify elastic alive by running the following in pod shell (MySql or elasticsearch):
curl -X GET -u elastic:admin1234 "http://<ClusterIP>:9200/_cluster/health?pretty=true"
curl -X GET -u elastic:admin1234 "http://10.100.217.227:9200/_cluster/health?pretty=true"
curl -X GET -u elastic:admin1234 "http://localhost:9200/_cluster/health?pretty=true"
